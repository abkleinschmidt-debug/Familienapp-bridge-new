  <script>

    // Erwartet Parameter: ?text=...

    // Beispiel: open.html?text=%F0%9F%8D%B3%20Archiv%20-%20Schnelle%20K%C3%BCche

    const params = new URLSearchParams(location.search);

    const text = params.get("text") || "";


    const info = document.getElementById("info");

    const btn  = document.getElementById("openBtn");


    if (!text) {

      info.textContent = "Fehlender Parameter: ?text=...";

      btn.style.opacity = "0.4";

      btn.onclick = (e) => { e.preventDefault(); alert("Kein Text übergeben."); };

    } else {

      info.innerHTML = "Ziel-Notiz: <code>" + decodeURIComponent(text) + "</code>";

      const shortcutName = "Notiz%20%C3%96ffnen"; // muss exakt so heißen wie dein Shortcut

      btn.href = "shortcuts://run-shortcut?name=" + shortcutName + "&input=text&text=" + text;

    }

  </script>
